//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    Looks.Hide();
    SCORE = 0;
    FRUIT GRID # = "";
    fruit placed = (1 == 0);
    fruit eaten = (1 == 0);
    SNAKE BLOCK SIZE = (BLOCK SIZE (PX) / 2);
    x direction = -1;
    y direction = 0;
    List.DeleteAll(Snake);
    List.DeleteAll(bottom edge);
    List.DeleteAll(top edge);
    List.DeleteAll(left edge);
    List.DeleteAll(right edge);
    Call Clear Grid;
    Call Initialise Edges;
    Call Initialise Snake;
    Event.Broadcast("Start Game Loop");
}


//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(Start Game Loop)
{
    Forever
    {
        Call Board Sequence;
        Call Place Fruit;
        Call Player Input;
        Call Check Game Over;
        Call Move Snake;
        Call Clear Grid;
        Control.Wait(.1);
    }
}

WhenBroadcastReceived(Stop Game Loop)
{
    Stop(all);
}


//----- Custom blocks -------------------------------------------------------------------------------------------------

Define Board Sequence (warp=true)
{
    Pen.Clear();
    Call Draw Snake;
    Call Draw Grid;
    Event.Broadcast("Draw Border");
}

Define Check Game Over (warp=true)
{
    l = 1;
    Repeat Until (l > List.Length(left edge))
    {
        If ((Snake[1] == left edge[l]) And (x direction == -1))
        {
            Event.BroadcastAndWait("Stop Game Loop");
        }
        Else
        {
            l += 1;
        }
    }
    l = 1;
    Repeat Until (l > List.Length(right edge))
    {
        If ((Snake[1] == right edge[l]) And (x direction == 1))
        {
            Event.BroadcastAndWait("Stop Game Loop");
        }
        Else
        {
            l += 1;
        }
    }
    l = 1;
    Repeat Until (l > List.Length(top edge))
    {
        If ((Snake[1] == top edge[l]) And (y direction == 1))
        {
            Event.BroadcastAndWait("Stop Game Loop");
        }
        Else
        {
            l += 1;
        }
    }
    l = 1;
    Repeat Until (l > List.Length(bottom edge))
    {
        If ((Snake[1] == bottom edge[l]) And (y direction == -1))
        {
            Event.BroadcastAndWait("Stop Game Loop");
        }
        Else
        {
            l += 1;
        }
    }
    tmp = "";
    l = 2;
    Repeat Until (l > List.Length(Snake))
    {
        tmp = Snake[l];
        If (Snake[1] == tmp)
        {
            Event.BroadcastAndWait("Stop Game Loop");
        }
        Else
        {
            l += 1;
        }
    }
}

Define Clear Grid (warp=true)
{
    List.DeleteAll(block list);
    Repeat (2296)
    {
        List.Add(block list, 0);
    }
    If (fruit placed == 1)
    {
        List.ReplaceItem(block list, FRUIT GRID #, 2);
    }
}

Define Draw Grid (warp=true)
{
    i = 1;
    Motion.SetY(-168);
    Repeat (41)
    {
        Motion.SetX(-228);
        Motion.ChangeYBy(8);
        Repeat (56)
        {
            Motion.ChangeXBy(8);
            If (block list[i] == 2)
            {
                FRUIT X = Motion.XPosition();
                FRUIT Y = Motion.YPosition();
                Event.Broadcast("Draw Fruit");
            }
            If (block list[i] == 1)
            {
                Pen.Stamp();
                i += 1;
            }
            Else
            {
                i += 1;
            }
        }
    }
}

Define Draw Snake (warp=true)
{
    tmp = "";
    j = 1;
    Repeat (List.Length(Snake))
    {
        tmp = Snake[j];
        List.ReplaceItem(block list, tmp, 1);
        j += 1;
    }
}

Define Initialise Edges (warp=true)
{
    k = 1;
    Repeat (56)
    {
        List.Add(bottom edge, k);
        k += 1;
    }
    k = 1;
    Repeat (41)
    {
        List.Add(left edge, k);
        k += 56;
    }
    k = 56;
    Repeat (41)
    {
        List.Add(right edge, k);
        k += 56;
    }
    k = 2241;
    Repeat (56)
    {
        List.Add(top edge, k);
        k += 1;
    }
}

Define Initialise Snake (warp=true)
{
    List.Add(Snake, 1148);
    List.Add(Snake, 1149);
    List.Add(Snake, 1150);
    List.Add(Snake, 1151);
}

Define Move Snake (warp=true)
{
    m = 1;
    tmp = "";
    Repeat Until (m == List.Length(Snake))
    {
        List.ReplaceItem(Snake, (List.Length(Snake) - (m - 1)), Snake[(List.Length(Snake) - m)]);
        m += 1;
    }
    If (y direction == 0)
    {
        If (x direction == -1)
        {
            List.ReplaceItem(Snake, 1, (Snake[1] + x direction));
        }
        Else
        {
            If (x direction == 1)
            {
                List.ReplaceItem(Snake, 1, (Snake[1] + x direction));
            }
        }
    }
    If (x direction == 0)
    {
        If (y direction == -1)
        {
            List.ReplaceItem(Snake, 1, (Snake[1] + (y direction - 55)));
        }
        Else
        {
            If (y direction == 1)
            {
                List.ReplaceItem(Snake, 1, (Snake[1] + (y direction + 55)));
            }
        }
    }
    tmp = Snake[1];
    If (block list[tmp] == 2)
    {
        fruit eaten = 1;
        fruit placed = 0;
        SCORE += 1;
        tmp = "";
    }
    If (fruit eaten == 1)
    {
        tmp = Snake[List.Length(Snake)];
        List.Add(Snake, tmp);
        fruit eaten = 0;
    }
}

Define Place Fruit (warp=true)
{
    n = 1;
    If (fruit placed == 0)
    {
        Repeat Until (fruit placed == 1)
        {
            FRUIT GRID # = Operator.Random(1 / List.Length(block list));
            If (Not ((block list[FRUIT GRID #] == 1)))
            {
                List.ReplaceItem(block list, FRUIT GRID #, 2);
                fruit placed = (1 == 1);
            }
        }
    }
}

Define Player Input (warp=true)
{
    If (Sensing.KeyPressed(left arrow) And (x direction == 0))
    {
        x direction = -1;
        y direction = 0;
    }
    If (Sensing.KeyPressed(right arrow) And (x direction == 0))
    {
        x direction = 1;
        y direction = 0;
    }
    If (Sensing.KeyPressed(down arrow) And (y direction == 0))
    {
        y direction = -1;
        x direction = 0;
    }
    If (Sensing.KeyPressed(up arrow) And (y direction == 0))
    {
        y direction = 1;
        x direction = 0;
    }
}
